-- Tabela DadosMotores
CREATE TABLE DadosMotores (
    id_motor SERIAL PRIMARY KEY,
    tag VARCHAR(255) NOT NULL,
    descricao VARCHAR(255) NOT NULL,
    frequencia_norm FLOAT NOT NULL,
    potencia_norm FLOAT NOT NULL,
    corrente_norm FLOAT NOT NULL,
    tensao_norm FLOAT NOT NULL,
    timestamp TIMESTAMP NOT NULL
);

-- Tabela DatalogMotores
CREATE TABLE DatalogMotores (
    id_log SERIAL PRIMARY KEY,
    id_motor INT NOT NULL,
    status BOOLEAN NOT NULL,
    spFrequencia FLOAT NOT NULL,
    feedCorrente FLOAT NOT NULL,
    feedTensaoEntrada FLOAT NOT NULL,
    timestamp TIMESTAMP NOT NULL,
    FOREIGN KEY (id_motor) REFERENCES DadosMotores(id_motor)
);

-- Tabela EventosVazao
CREATE TABLE EventosVazao (
    id_evento SERIAL PRIMARY KEY,
    id_log INT NOT NULL,
    vazaoRegistrada FLOAT NOT NULL,
    timestamp TIMESTAMP NOT NULL,
    FOREIGN KEY (id_log) REFERENCES DatalogMotores(id_log)
);

-- Tabela previsaoM3h 
CREATE TABLE previsaoM3h (
    id_previsao SERIAL PRIMARY KEY,
    previsaoRegistrada FLOAT NOT NULL,
    offsetTolerancia FLOAT NOT NULL,
    timestamp TIMESTAMP NOT NULL
);
